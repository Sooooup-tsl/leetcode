/**
 * @param {number[]} nums
 * @param {number} k
 * @return {boolean}
 */
var containsNearbyDuplicate = function(nums, k) {
    let set= new Map()
    for(var i = 0;i<nums.length;i++) {
        if (set.has(nums[i]) && (i-set.get(nums[i]) <= k)){
            return true
        }

        set.set(nums[i], i)
    }

    return false
};
